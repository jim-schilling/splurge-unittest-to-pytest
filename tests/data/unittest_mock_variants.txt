from unittest.mock import patch, MagicMock
from unittest import mock

class TestMockVariants(unittest.TestCase):
    def setUp(self):
        self.m = MagicMock()

    def test_patch(self):
        with patch('os.path.exists') as p:
            p.return_value = True
            self.assertTrue(p('x'))

    def test_mock_module(self):
        m = mock.Mock()
        m.side_effect = ValueError('boom')
        with self.assertRaises(ValueError):
            m()
