class TestInheritanceChain:
    """Test case combining API and database setup."""

    def test_inheritance_chain(self, full_stack_setup):
        """Test that all setup methods in inheritance chain work."""
        full_stack_resource, api_client, db_connection, db_cursor = full_stack_setup

        # Verify class-level resources
        assert full_stack_resource == "combined_resource"
        assert api_client is not None
        assert db_connection is not None

    def test_api_database_integration(self, full_stack_setup):
        """Test API and database working together."""
        full_stack_resource, api_client, db_connection, db_cursor = full_stack_setup

        # Simulate API call that triggers database operation
        api_client.get.return_value = {"user_id": 1, "data": "test"}

        # Simulate database query
        db_cursor.fetchall.return_value = [{"id": 1, "name": "Test User"}]

        # Verify both systems work
        api_result = api_client.get("/users/1")
        db_result = db_cursor.fetchall()

        assert api_result["user_id"] == 1
        assert len(db_result) == 1
        assert db_result[0]["name"] == "Test User"

    def test_resource_cleanup(self, base_instance_setup):
        """Test that resources are properly cleaned up."""
        base_setup_called, base_counter, base_resources, logger = base_instance_setup

        # Add a resource that needs cleanup
        mock_resource = MagicMock()
        base_resources.append(mock_resource)
