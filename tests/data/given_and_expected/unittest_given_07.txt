import unittest


class TestBasicIO(unittest.TestCase):
    def setUp(self):
        self.file_handler = FileHandler()
        self.test_content = "Hello, World!"
    
    def tearDown(self):
        self.file_handler = None
        self.test_content = None
    
    def test_read_content(self):
        content = self.file_handler.read_string(self.test_content)
        self.assertEqual(content, "Hello, World!")
    
    def test_content_length(self):
        length = self.file_handler.get_length(self.test_content)
        self.assertEqual(length, 13)


class FileHandler:
    def read_string(self, content):
        return content
    
    def get_length(self, content):
        return len(content)
