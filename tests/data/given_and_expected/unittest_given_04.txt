import unittest


class TestListOperations(unittest.TestCase):
    def setUp(self):
        self.numbers = [1, 2, 3, 4, 5]
        self.list_processor = ListProcessor()
    
    def tearDown(self):
        self.numbers = None
        self.list_processor = None
    
    def test_sum_all(self):
        result = self.list_processor.sum_all(self.numbers)
        self.assertEqual(result, 15)
    
    def test_find_max(self):
        result = self.list_processor.find_max(self.numbers)
        self.assertEqual(result, 5)
    
    def test_filter_even(self):
        result = self.list_processor.filter_even(self.numbers)
        self.assertEqual(result, [2, 4])


class ListProcessor:
    def sum_all(self, numbers):
        return sum(numbers)
    
    def find_max(self, numbers):
        return max(numbers)
    
    def filter_even(self, numbers):
        return [n for n in numbers if n % 2 == 0]
