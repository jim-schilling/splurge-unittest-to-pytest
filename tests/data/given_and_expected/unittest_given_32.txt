import unittest
import pytest
from unittest import mock

class BaseTest(unittest.TestCase):
    def helper(self, x):
        return x * 2

class TestMultiInherit(BaseTest, unittest.TestCase):
    def test_helper(self):
        self.assertEqual(self.helper(3), 6)

    def test_using_mock(self):
        m = mock.Mock()
        m.method.return_value = 42
        self.assertEqual(m.method(), 42)

    def test_with_subtests(self):
        for i in range(3):
            with self.subTest(i=i):
                self.assertTrue(i < 5)


def test_already_pytest_style():
    assert 1 + 1 == 2

if __name__ == '__main__':
    unittest.main()
