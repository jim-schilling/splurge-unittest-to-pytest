class TestAsyncSetupTeardown:
    """Test case with asynchronous setup and teardown operations."""

    def test_thread_safety(self, thread_management):
        """Test thread-safe operations."""
        threads, thread_results, shared_resource = thread_management

        def worker_thread(thread_id):
            """Worker function for thread testing."""
            with shared_resource:
                thread_results[thread_id] = f"completed_{thread_id}"
                time.sleep(0.1)  # Simulate work

        # Start multiple threads
        for i in range(3):
            thread = threading.Thread(target=worker_thread, args=(i,))
            threads.append(thread)
            thread.start()

        # Wait for threads to complete
        for thread in threads:
            thread.join(timeout=2.0)

        # Verify results
        assert len(thread_results) == 3
        for i in range(3):
            assert i in thread_results
            assert thread_results[i] == f"completed_{i}"

    def test_thread_local_storage(self, thread_synchronization):
        """Test thread-local data isolation."""
        thread_id, start_time, setup_events, local_data, resource_acquired = thread_synchronization

        results = {}

        def check_local_data(thread_id):
            """Check that thread-local data is isolated."""
            local = threading.local()
            local.test_value = f"thread_{thread_id}"
            time.sleep(0.05)  # Allow context switching
            results[thread_id] = local.test_value

        # Start threads with different local data
        threads = []
        for i in range(2):
            thread = threading.Thread(target=check_local_data, args=(i,))
            threads.append(thread)
            thread.start()

        # Wait for completion
        for thread in threads:
            thread.join()

        # Verify isolation
        assert results[0] == "thread_0"
        assert results[1] == "thread_1"
        assert results[0] != results[1]
