import unittest
from unittest.mock import Mock, patch, MagicMock
import threading
import time
import queue

class TestConcurrentProcessor(unittest.TestCase):
    def setUp(self):
        self.input_queue = queue.Queue()
        self.output_queue = queue.Queue()
        self.error_queue = queue.Queue()
        
        # Add test data to input queue
        self.test_items = [
            {"id": 1, "data": "item1"},
            {"id": 2, "data": "item2"},
            {"id": 3, "data": "item3"}
        ]
        
        for item in self.test_items:
            self.input_queue.put(item)
            
        # Mock external services
        self.mock_processor = Mock()
        self.mock_validator = Mock()
        self.mock_database = Mock()
        
        # Threading components
        self.worker_threads = []
        self.stop_event = threading.Event()
        
    def tearDown(self):
        # Signal all threads to stop
        self.stop_event.set()
        
        # Wait for all threads to finish
        for thread in self.worker_threads:
            if thread.is_alive():
                thread.join(timeout=1.0)
                
        # Clear all queues
        self._clear_queue(self.input_queue)
        self._clear_queue(self.output_queue)
        self._clear_queue(self.error_queue)
        
    def _clear_queue(self, q):
        try:
            while True:
                q.get_nowait()
        except queue.Empty:
            pass
            
    @patch('time.sleep')
    def test_single_item_processing(self, mock_sleep):
        mock_sleep.return_value = None  # Speed up test
        self.mock_processor.process.return_value = {"processed": True}
        self.mock_validator.validate.return_value = True
        
        # Test single item processing
        item = self.test_items[0]
        
        # Simulate processing
        validated = self.mock_validator.validate(item)
        self.assertTrue(validated)
        
        result = self.mock_processor.process(item)
        self.assertIn("processed", result)
        self.assertTrue(result["processed"])
        
    def test_queue_operations(self):
        # Test queue has items
        self.assertFalse(self.input_queue.empty())
        self.assertEqual(self.input_queue.qsize(), 3)
        
        # Test getting items
        item = self.input_queue.get()
        self.assertEqual(item["id"], 1)
        self.assertEqual(self.input_queue.qsize(), 2)
        
        # Test putting items in output queue
        processed_item = {"id": item["id"], "status": "completed"}
        self.output_queue.put(processed_item)
        self.assertEqual(self.output_queue.qsize(), 1)
        
    @patch('threading.Thread')
    def test_worker_thread_creation(self, mock_thread_class):
        mock_thread = Mock()
        mock_thread_class.return_value = mock_thread
        
        # Simulate thread creation
        thread = mock_thread_class(target=lambda: None, daemon=True)
        self.worker_threads.append(thread)
        
        self.assertEqual(len(self.worker_threads), 1)
        mock_thread_class.assert_called_once()
