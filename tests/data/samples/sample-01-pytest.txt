import pytest
import tempfile
import shutil
from pathlib import Path
from typing import Generator

@pytest.fixture
def temp_dir() -> Generator[Path, None, None]:
    """Create temporary directory for testing."""
    temp_dir = Path(tempfile.mkdtemp())
    try:
        yield temp_dir
    finally:
        shutil.rmtree(temp_dir, ignore_errors=True)

@pytest.fixture
def test_file(temp_dir: Path) -> Path:
    """Create test file path in temporary directory."""
    return temp_dir / "test.txt"

def test_file_creation(test_file: Path) -> None:
    """Test file creation functionality."""
    test_file.write_text("Hello World")
    assert test_file.exists()

def test_file_content(test_file: Path) -> None:
    """Test file content operations."""
    content = "Test content"
    test_file.write_text(content)
    assert test_file.read_text() == content
