import unittest
from unittest.mock import Mock, patch, MagicMock
import requests
import time
import os
from contextlib import contextmanager

class TestExternalServiceIntegration(unittest.TestCase):
    
    @classmethod
    def setUpClass(cls):
        cls.api_base_url = os.getenv('API_BASE_URL', 'https://api.example.com')
        cls.api_token = os.getenv('API_TOKEN', 'test_token')
        cls.timeout = int(os.getenv('API_TIMEOUT', '30'))
        
        # Mock service registry
        cls.service_registry = {
            'auth_service': f'{cls.api_base_url}/auth',
            'user_service': f'{cls.api_base_url}/users',
            'data_service': f'{cls.api_base_url}/data'
        }
        
    def setUp(self):
        self.session = requests.Session()
        self.session.headers.update({
            'Authorization': f'Bearer {self.api_token}',
            'Content-Type': 'application/json'
        })
        
        # Mock responses
        self.mock_auth_response = {
            'token': 'mock_auth_token',
            'expires_in': 3600,
            'user_id': 'test_user_123'
        }
        
        self.mock_user_data = {
            'id': 'test_user_123',
            'username': 'testuser',
            'email': 'test@example.com',
            'created_at': '2024-01-01T00:00:00Z'
        }
        
        # Performance tracking
        self.performance_data = []
        
    def tearDown(self):
        self.session.close()
        
        # Log performance data if any tests measured timing
        if self.performance_data:
            avg_response_time = sum(self.performance_data) / len(self.performance_data)
            print(f"Average response time: {avg_response_time:.3f}s")
            
    @contextmanager
    def measure_time(self):
        """Context manager to measure execution time."""
        start_time = time.time()
        try:
            yield
        finally:
            end_time = time.time()
            self.performance_data.append(end_time - start_time)
            
    def mock_successful_response(self, data, status_code=200):
        """Helper to create mock response."""
        mock_response = Mock()
        mock_response.json.return_value = data
        mock_response.status_code = status_code
        mock_response.raise_for_status.return_value = None
        return mock_response
        
    def mock_error_response(self, status_code, error_message):
        """Helper to create mock error response."""
        mock_response = Mock()
        mock_response.status_code = status_code
        mock_response.json.return_value = {'error': error_message}
        mock_response.raise_for_status.side_effect = requests.HTTPError(error_message)
        return mock_response
        
    @patch('requests.Session.post')
    def test_authentication_flow(self, mock_post):
        mock_post.return_value = self.mock_successful_response(self.mock_auth_response)
        
        with self.measure_time():
            # Simulate authentication
            auth_url = self.service_registry['auth_service']
            response = self.session.post(auth_url, json={
                'username': 'testuser',
                'password': 'testpass'
            })
            
        self.assertEqual(response.status_code, 200)
        auth_data = response.json()
        self.assertIn('token', auth_data)
        self.assertEqual(auth_data['user_id'], 'test_user_123')
        
        # Verify the request was made correctly
        mock_post.assert_called_once()
        call_args = mock_post.call_args
        self.assertEqual(call_args[0][0], auth_url)
        
    @patch('requests.Session.get')
    @patch('requests.Session.post')
    def test_user_data_retrieval_after_auth(self, mock_post, mock_get):
        # Setup auth mock
        mock_post.return_value = self.mock_successful_response(self.mock_auth_response)
        
        # Setup user data mock
        mock_get.return_value = self.mock_successful_response(self.mock_user_data)
        
        # First authenticate
        auth_response = self.session.post(
            self.service_registry['auth_service'],
            json={'username': 'testuser', 'password': 'testpass'}
        )
        self.assertEqual(auth_response.status_code, 200)
        
        # Then get user data
        user_id = auth_response.json()['user_id']
        user_response = self.session.get(
            f"{self.service_registry['user_service']}/{user_id}"
        )
        
        self.assertEqual(user_response.status_code, 200)
        user_data = user_response.json()
        self.assertEqual(user_data['id'], user_id)
        self.assertEqual(user_data['username'], 'testuser')
        
    @patch('requests.Session.get')
    def test_service_error_handling(self, mock_get):
        # Test different error scenarios
        error_scenarios = [
            (404, 'User not found'),
            (500, 'Internal server error'),
            (403, 'Insufficient permissions')
        ]
        
        for status_code, error_message in error_scenarios:
            with self.subTest(status_code=status_code):
                mock_get.return_value = self.mock_error_response(status_code, error_message)
                
                response = self.session.get(f"{self.service_registry['user_service']}/invalid")
                
                self.assertEqual(response.status_code, status_code)
                with self.assertRaises(requests.HTTPError):
                    response.raise_for_status()
                    
    @patch('requests.Session.get')
    def test_timeout_handling(self, mock_get):
        mock_get.side_effect = requests.Timeout("Request timed out")
        
        with self.assertRaises(requests.Timeout):
            self.session.get(self.service_registry['data_service'], timeout=self.timeout)
            
    @patch('time.sleep')  # Speed up retry logic in tests
    @patch('requests.Session.get')
    def test_retry_mechanism(self, mock_get, mock_sleep):
        mock_sleep.return_value = None
        
        # First two calls fail, third succeeds
        mock_get.side_effect = [
            requests.ConnectionError("Connection failed"),
            requests.ConnectionError("Connection failed"),
            self.mock_successful_response({'status': 'success'})
        ]
        
        # Simulate retry logic (would be in actual application code)
        max_retries = 3
        for attempt in range(max_retries):
            try:
                response = self.session.get(self.service_registry['data_service'])
                break
            except requests.ConnectionError:
                if attempt == max_retries - 1:
                    raise
                time.sleep(2 ** attempt)  # Exponential backoff
                
        self.assertEqual(response.status_code, 200)
        self.assertEqual(mock_get.call_count, 3)
