    var_a: str
    var_b: str

    """Set up test resources and data.
    
    Returns:
        _TestInitApiData containing resource, content, and derived variables.
    """
    resource = util.get_resource()
    content = """String Literal"""
    
    try:

            content

        util.cleanup(resource)
