import unittest
from parameterized import parameterized

class ExpectedFailureParam(unittest.TestCase):

    @parameterized.expand([(1,), (2,), (3,)])
    @unittest.expectedFailure
    def test_param_expected(self, value):
        # expected failure depending on environment
        self.assertEqual(value, 0)

    @unittest.expectedFailure
    @parameterized.expand([(True,), (False,)])
    def test_expected_and_param_second(self, flag):
        self.assertTrue(flag)
