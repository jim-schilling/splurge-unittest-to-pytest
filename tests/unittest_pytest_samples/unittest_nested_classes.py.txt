import unittest


class OuterTest(unittest.TestCase):
    def setUp(self) -> None:
        self.a = 1

    class Inner(unittest.TestCase):
        def setUp(self) -> None:
            self.b = 2

        def test_inner(self) -> None:
            assert self.b == 2

    def test_outer(self) -> None:
        assert self.a == 1
