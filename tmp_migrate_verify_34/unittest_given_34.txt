import pytest


class TestContextManager:
    def test_open_file(self):
        with open(__file__, "r") as f:
            content = f.read()
            assert len(content) > 0

    def test_subtest_and_nested(self, subtests):
        with subtests.test(msg="outer"):
            with subtests.test(msg="inner"):
                assert 1 is not None


@pytest.mark.parametrize("a,b,exp", [(1, 2, 3), (2, 3, 5)])
def test_param(a, b, exp):
    assert a + b == exp
