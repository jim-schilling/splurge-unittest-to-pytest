import os

import pytest


class TestSkipAndClasssetup:
    @pytest.fixture(scope="class", autouse=True)
    def setup_class(cls):
        cls.tmpdir = "tmpdir"
        yield

    @pytest.mark.skipif(os.name == "nt", "Not on Windows")
    def test_skipped_on_windows(self):
        assert True

    def test_raises_regex(self):
        with pytest.raises(ValueError):
            raise ValueError("bad value")


def test_mixed_asserts_and_pytest():
    assert "a" in "abc"
    assert not False
